106template<typename Value_>
107void compare_vectors(const std::vector<Value_>& expected, const std::vector<Value_>& observed, const std::string& context) {
108 size_t n_expected = expected.size();
109 ASSERT_EQ(n_expected, observed.size()) << "mismatch in vector length (" << context << ")";
110 for (size_t i = 0; i < n_expected; ++i) {
111 auto expected_val = expected[i], observed_val = observed[i];
112 if (std::isnan(expected_val)) {
113 EXPECT_EQ(std::isnan(expected_val), std::isnan(observed_val)) << "mismatching NaNs at position " << i << " (" << context << ")";
114 } else {
115 EXPECT_EQ(expected_val, observed_val) << "different values at position " << i << " (" << context << ")";
116 }
117 }
118}

121uint64_t create_seed(Index_ NR, Index_ NC, const TestAccessOptions& options) {
122 uint64_t seed = static_cast<uint64_t>(NR) * static_cast<uint64_t>(NC);
123 seed += 13 * static_cast<uint64_t>(options.use_row);
124 seed += 57 * static_cast<uint64_t>(options.order);
125 seed += 101 * static_cast<uint64_t>(options.jump);

129template<typename Index_>
130std::vector<Index_> simulate_test_access_sequence(Index_ NR, Index_ NC, const TestAccessOptions& options) {
131 std::vector<Index_> sequence;
132 auto limit = (options.use_row ? NR : NC);
133
134 if (options.order == TestAccessOrder::REVERSE) {
135 for (int i = limit; i > 0; i -= options.jump) {
136 sequence.push_back(i - 1);
137 }
138 } else {
139 for (int i = 0; i < limit; i += options.jump) {
140 sequence.push_back(i);
141 }
142 if (options.order == TestAccessOrder::RANDOM) {
143 std::mt19937_64 rng(create_seed(NR, NC, options));
144 std::shuffle(sequence.begin(), sequence.end(), rng);
145 }
146 }
147
148 return sequence;
149}

151template<bool use_oracle_, typename Index_>
152tatami::MaybeOracle<use_oracle_, Index_> create_oracle(const std::vector<Index_>& sequence, const TestAccessOptions& options) {
153 if constexpr(use_oracle_) {
154 std::shared_ptr<tatami::Oracle<Index_> > oracle;
155 if (options.jump == 1 && options.order == TestAccessOrder::FORWARD) {
156 oracle.reset(new tatami::ConsecutiveOracle<Index_>(0, sequence.size()));
157 } else {
158 oracle.reset(new tatami::FixedViewOracle<Index_>(sequence.data(), sequence.size()));
159 }
160 return oracle;
161 } else {
162 return false;
163 }

166template<bool use_oracle_, typename Value_, typename Index_, class SparseExpand_, typename ...Args_>
167void test_access_base(
168 const tatami::Matrix<Value_, Index_>& matrix,
169 const tatami::Matrix<Value_, Index_>& reference,
170 const TestAccessOptions& options,
171 Index_ extent,
172 SparseExpand_ sparse_expand,
173 Args_... args)
174{
175 auto NR = matrix.nrow();
176 ASSERT_EQ(NR, reference.nrow());
177 auto NC = matrix.ncol();
178 ASSERT_EQ(NC, reference.ncol());
179
180 auto refwork = (options.use_row ? reference.dense_row(args...) : reference.dense_column(args...));
181
182 auto sequence = simulate_test_access_sequence(NR, NC, options);
183 auto oracle = create_oracle<use_oracle_>(sequence, options);
184
185 auto pwork = tatami::new_extractor<false, use_oracle_>(&matrix, options.use_row, oracle, args...);
186 auto swork = tatami::new_extractor<true, use_oracle_>(&matrix, options.use_row, oracle, args...);
187
188 tatami::Options opt;
189 opt.sparse_extract_index = false;
190 auto swork_v = tatami::new_extractor<true, use_oracle_>(&matrix, options.use_row, oracle, args..., opt);
191
192 opt.sparse_extract_value = false;
193 auto swork_n = tatami::new_extractor<true, use_oracle_>(&matrix, options.use_row, oracle, args..., opt);
194
195 opt.sparse_extract_index = true;
196 auto swork_i = tatami::new_extractor<true, use_oracle_>(&matrix, options.use_row, oracle, args..., opt);
197
198 size_t sparse_counter = 0;
199
200 // Looping over rows/columns and checking extraction against the reference.
201 for (auto i : sequence) {
202 auto expected = fetch(*refwork, i, extent);
203
204 // Checking dense retrieval first.
205 {
206 auto observed = [&]() {
207 if constexpr(use_oracle_) {
208 return fetch(*pwork, extent);
209 } else {
210 return fetch(*pwork, i, extent);
211 }
212 }();
213 compare_vectors(expected, observed, "dense retrieval");
214 }
215
216 // Various flavors of sparse retrieval.
217 {
218 auto observed = [&]() {
219 if constexpr(use_oracle_) {
220 return fetch(*swork, extent);
221 } else {
222 return fetch(*swork, i, extent);
223 }
224 }();
225 compare_vectors(expected, sparse_expand(observed), "sparse retrieval");
226
227 sparse_counter += observed.value.size();
228 {
229 bool is_increasing = true;
230 for (size_t i = 1; i < observed.index.size(); ++i) {
231 if (observed.index[i] <= observed.index[i-1]) {
232 is_increasing = false;
233 break;
234 }
235 }
236 ASSERT_TRUE(is_increasing);
237 }
238
239 std::vector<Index_> indices(extent);
240 auto observed_i = [&]() {
241 if constexpr(use_oracle_) {
242 return swork_i->fetch(NULL, indices.data());
243 } else {
244 return swork_i->fetch(i, NULL, indices.data());
245 }
246 }();
247 ASSERT_TRUE(observed_i.value == NULL);
248 tatami::copy_n(observed_i.index, observed_i.number, indices.data());
249 indices.resize(observed_i.number);
250 ASSERT_EQ(observed.index, indices);
251
252 std::vector<Value_> values(extent);
253 auto observed_v = [&]() {
254 if constexpr(use_oracle_) {
255 return swork_v->fetch(values.data(), NULL);
256 } else {
257 return swork_v->fetch(i, values.data(), NULL);
258 }
259 }();
260 ASSERT_TRUE(observed_v.index == NULL);
261 tatami::copy_n(observed_v.value, observed_v.number, values.data());
262 values.resize(observed_v.number);
263 compare_vectors(values, observed.value, "sparse retrieval with values only");
264
265 auto observed_n = [&]() {
266 if constexpr(use_oracle_) {
267 return swork_n->fetch(NULL, NULL);
268 } else {
269 return swork_n->fetch(i, NULL, NULL);
270 }
271 }();
272 ASSERT_TRUE(observed_n.value == NULL);
273 ASSERT_TRUE(observed_n.index == NULL);
274 ASSERT_EQ(observed.value.size(), observed_n.number);
275 }
276 }
277
278 if (options.check_sparse && matrix.is_sparse()) {
279 EXPECT_TRUE(sparse_counter < static_cast<size_t>(NR) * static_cast<size_t>(NC));
280 }
281}

335template<typename Index_>
336std::vector<Index_> create_indexed_subset(Index_ nsecondary, double relative_start, double probability, uint64_t base_seed) {
337 Index_ start = nsecondary * relative_start;
338 std::vector<Index_> indices { start };
339 std::mt19937_64 rng(base_seed + 999 * probability + 888 * start);
340 std::uniform_real_distribution udist;
341 for (Index_ i = start + 1; i < nsecondary; ++i) {
342 if (udist(rng) < probability) {
343 indices.push_back(i);
344 }
345 }
346 return indices;
347}
